-----------MYSQL JOINS---------

----INNER JOIN
--------------------------------------------------------------------------
//1.WRITE A QUERY TO RETRIVE THE DATA FROM OFFICERS TABLE AND THE CORRESPONDING COURSE NAME FROM STUDENTS TABLE 
USING INNER JOIN.

SELECT OFFICERS.OFFICER_NAME,OFFICERS.ADDRESS,STUDENTS.COURSE_NAME
FROM OFFICERS 
INNER JOIN STUDENTS 
ON OFFICERS.OFFICER_ID=STUDENTS.STUDENT_ID;

//2.WRITE A QUERY TO RETRIVE THE DATA FROM STUDENTS TABLE AND THE CORRESPONDING MARKS FROM MARKS TABLE 
USING INNER JOIN.

SELECT STUDENTS.STUDENT_ID,STUDENTS.STUDENT_NAME,STUDENTS.COURSE_NAME,MARKS.MARKS
FROM STUDENTS INNER JOIN MARKS
ON STUDENTS.STUDENT_ID=MARKS.STUDENT_ID;

---LEFT OUTER JOIN
---------------------------------------------------------------------------------------
//3.WRITE A QUERY TO RETRIVE DATA FROM STUDENTS TABLE AND MARKS TABLE USING LEFT OUTER JOIN.

SELECT STUDENTS.STUDENT_ID,STUDENTS.STUDENT_NAME,STUDENTS.COURSE_NAME,MARKS.MARKS
FROM STUDENTS LEFT OUTER JOIN MARKS
ON STUDENTS.STUDENT_ID=MARKS.STUDENT_ID;

//4.WRITE A QUERY TO RETRIVE DATA FROM OFFICERS TABLE AND STUDENTS TABLE USING LEFT OUTER JOIN.

SELECT OFFICERS.OFFICER_NAME,OFFICERS.ADDRESS,STUDENTS.COURSE_NAME
FROM OFFICERS LEFT JOIN STUDENTS 
ON OFFICERS.OFFICER_ID=STUDENTS.STUDENT_ID;

---RIGHT OUTER JOIN
------------------------------------------------------------------------------------------------
//5.WRITE A QUERY TO RETRIVE THE DATA FROM STUDENTS TABLE AND THE CORRESPONDING MARKS FROM MARKS TABLE 
USING RIGHT OUTER JOIN.

SELECT STUDENTS.STUDENT_ID,STUDENTS.STUDENT_NAME,STUDENTS.COURSE_NAME,MARKS.MARKS
FROM STUDENTS RIGHT OUTER JOIN MARKS
ON STUDENTS.STUDENT_ID=MARKS.STUDENT_ID;

//6.WRITE A QUERY TO RETRIVE THE DATA FROM STUDENTS TABLE AND THE CORRESPONDING MARKS FROM MARKS TABLE 
USING RIGHT OUTER JOIN.

SELECT OFFICERS.OFFICER_NAME,OFFICERS.ADDRESS,STUDENTS.COURSE_NAME
FROM OFFICERS RIGHT JOIN STUDENTS 
ON OFFICERS.OFFICER_ID=STUDENTS.STUDENT_ID;
